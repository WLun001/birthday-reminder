170       * helper method to show quote of the day dialog 
171       */ 
172      private void showQuoteDialog() { 
173          if(quotes == null){ 
174              Toast.makeText(this, R.string.no_quote_found, Toast.LENGTH_SHORT).show(); 
175              return; 
176          } 
177          int max = quotes.size(); 
178          final Quote quote = quotes.get(new Random().nextInt(max)); // ((max - min) + 1) + min 
179          AlertDialog.Builder builder = new AlertDialog.Builder(this); 
180          builder.setMessage(quote.getQuote()).setTitle(quote.getAuthor()) 
181                  .setPositiveButton(R.string.btn_ok, new DialogInterface.OnClickListener() { 
182                      @Override 
183                      public void onClick(DialogInterface dialog, int which) { 
184                          dialog.dismiss(); 
185                      } 
186                  }) 
187                  .setNeutralButton(R.string.dialog_copy_text, new DialogInterface.OnClickListener() { 
188                      @Override 
189                      public void onClick(DialogInterface dialog, int which) { 
190                          ClipboardManager clipboard = (ClipboardManager) getSystemService(Context.CLIPBOARD_SERVICE); 
191                          ClipData clip = ClipData.newPlainText(getString(R.string.dialog_copy_text), quote.getQuote()); 
192                          clipboard.setPrimaryClip(clip); 
193                          Toast.makeText(MainActivity.this, getString(R.string.copied_success), Toast.LENGTH_SHORT).show(); 
194                      } 
195                  }); 
196          AlertDialog dialog = builder.create(); 
197          dialog.show(); 
198      }

222      /** 
223       * helper method to init the notification service 
224       */ 
225      private void startNotification() { 
226          AlarmManager alarmManager; 
227   
228          Calendar calendar = Calendar.getInstance(); 
229          Calendar now = Calendar.getInstance(); 
230          Date date = new Date(); 
231          now.setTime(date); 
232          calendar.setTimeInMillis(System.currentTimeMillis()); 
233          calendar.set(Calendar.HOUR_OF_DAY, 8); 
234          calendar.set(Calendar.MINUTE, 00); 
235          calendar.set(Calendar.SECOND, 00); 
236   
237          Log.v(LOG_TAG, "Time: " + calendar.getTimeInMillis() + ""); 
238   
239          if (calendar.before(now)) { 
240              calendar.add(Calendar.DATE, 1); 
241          } 
242   
243          Intent intent = new Intent(this, NotiReceiver.class); 
244          PendingIntent alarmIntent = PendingIntent.getBroadcast(this, 0, intent, PendingIntent.FLAG_UPDATE_CURRENT); 
245          alarmManager = (AlarmManager) getSystemService(Context.ALARM_SERVICE); 
246          alarmManager.setRepeating(AlarmManager.RTC_WAKEUP, calendar.getTimeInMillis(), AlarmManager.INTERVAL_DAY, alarmIntent); 
247   
248          Log.v(LOG_TAG, "Alarm started"); 
249      }

269      /** 
270       * helper mehtod do perform backup to cloud 
271       */ 
272      private void backupToCloud() { 
273          ConnectivityManager connMgr = (ConnectivityManager) getSystemService(Context.CONNECTIVITY_SERVICE); 
274          NetworkInfo networkInfo = connMgr.getActiveNetworkInfo(); 
275          if (networkInfo != null && networkInfo.isConnected()) { 
276              getLoaderManager().restartLoader(BACKUP_LOADER_ID, null, this).forceLoad(); 
277          } else { 
278              Toast.makeText(this, getString(R.string.no_network), Toast.LENGTH_SHORT).show(); 
279          } 
280      }

288      @Override 
289      public void onLoadFinished(Loader<JSONObject> loader, JSONObject data) { 
290          Loader<JSONObject> backupLoader = getLoaderManager().getLoader(BACKUP_LOADER_ID); 
291          BackupLoader backupLoader1 = (BackupLoader) backupLoader; 
292          if (backupLoader1.progressDialogIsShow()) 
293              backupLoader1.stopProgressDialog(); 
294          AlertDialog alertDialog = new AlertDialog.Builder(this).create(); 
295          alertDialog.setMessage(String.format(getString(R.string.backup_success) 
296                  , extraCodeFromJSON(data))); 
297          alertDialog.setButton(AlertDialog.BUTTON_NEUTRAL, getString(R.string.btn_ok), 
298                  new DialogInterface.OnClickListener() { 
299                      public void onClick(DialogInterface dialog, int which) { 
300                          dialog.dismiss(); 
301                      } 
302                  }); 
303          alertDialog.show(); 
304      }
