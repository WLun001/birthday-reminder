32   public class BackupLoader extends android.content.AsyncTaskLoader<JSONObject> {

70       @Override 
71       public JSONObject loadInBackground() { 
72           Log.v(LOG_TAG, "BackupLoader doInBackground()"); 
73           JSONObject jsonRespose = null; 
74           try { 
75               readFromDb(); 
76               jsonRespose = postJSON(); 
77           } catch (IOException e) { 
78               Log.e(LOG_TAG, e.getMessage()); 
79           } 
80           return jsonRespose; 
81       }

83       /** 
84        * helper method to query data from database 
85        */ 
86       private void readFromDb() { 
87           PersonDBQueries dbQuery = new PersonDBQueries(new PersonDBHelper(getContext())); 
88           String[] columns = PersonContract.columns; 
89           Cursor cursor = dbQuery.query(columns, null, null, null, null 
90                   , PersonContract.PersonEntry.COLUMN_NAME_NAME + " ASC"); 
91           persons = PersonDBQueries.getPersonList(cursor); 
92       }

94       /** 
95        * helper mehtod to post JSONObject 
96        * 
97        * @return JSONObject that response from server 
98        * @throws IOException 
99        */ 
100      private JSONObject postJSON() throws IOException { 
101          InputStream inputStream = null; 
102          URL url = new URL(JSON_URL); 
103   
104          HttpURLConnection connection = (HttpURLConnection) url.openConnection(); 
105          connection.setReadTimeout(10000); 
106          connection.setConnectTimeout(15000); 
107          connection.setRequestMethod("POST"); 
108          connection.setDoInput(true); 
109   
110          try { 
111              OutputStream outputStream = connection.getOutputStream(); 
112              BufferedWriter writer = new BufferedWriter(new OutputStreamWriter(outputStream, "UTF-8")); 
113              writer.write(getPostDataString(putAsJSONArray())); 
114              writer.flush(); 
115              writer.close(); 
116              outputStream.close(); 
117   
118              int responseCode = connection.getResponseCode(); 
119              if (responseCode == 200) { 
120                  inputStream = connection.getInputStream(); 
121   
122                  // Convert the InputStream into ArrayList<Person> 
123                  return readInputStream(inputStream); 
124              } else { 
125                  Log.e(LOG_TAG, "ERROR:" + Integer.toString(responseCode)); 
126                  return null; 
127              } 
128          } catch (Exception e) { 
129              Log.e(LOG_TAG, e.getMessage()); 
130              return null; 
131          } finally { 
132              inputStream.close(); 
133          } 
134      }

136      /** 
137       * helper methhod to put person as JSONArray 
138       * 
139       * @return JSONArray 
140       */ 
141      private JSONArray putAsJSONArray() { 
142          JSONArray jsonArray = new JSONArray(); 
143          try { 
144              for (int i = 0; i < persons.size(); i++) { 
145                  jsonArray.put( 
146                          new JSONObject().put("id", persons.get(i).getId()) 
147                                  .put("name", persons.get(i).getName()) 
148                                  .put("email", persons.get(i).getEmail()) 
149                                  .put("phone", persons.get(i).getPhone()) 
150                                  .put("dob", persons.get(i).getDOB()) 
151                                  .put("notify", persons.get(i).isNotify()) 
152                  ); 
153              } 
154          } catch (JSONException e) { 
155              Log.e(LOG_TAG, e.getMessage()); 
156          } 
157          return jsonArray; 
158      }

160      /** 
161       * helper method to build string from JSONArray 
162       * 
163       * @param data 
164       * @return postDataString 
165       * @throws Exception 
166       */ 
167      private String getPostDataString(JSONArray data) throws Exception { 
168   
169          StringBuilder result = new StringBuilder(); 
170   
171          result.append(URLEncoder.encode("data", "UTF-8")); 
172          result.append("="); 
173          result.append(URLEncoder.encode(data.toString(), "UTF-8")); 
174   
175          return result.toString(); 
176      }

178      /** 
179       * helper method to read input stream 
180       * 
181       * @param is 
182       * @return JSONObject 
183       * @throws IOException 
184       * @throws JSONException 
185       */ 
186      private JSONObject readInputStream(InputStream is) 
187              throws IOException, JSONException { 
188          BufferedReader reader = new BufferedReader(new InputStreamReader(is, "UTF-8")); 
189          StringBuilder builder = new StringBuilder(); 
190   
191          String input; 
192          while ((input = reader.readLine()) != null) 
193              builder.append(input); 
194   
195          return new JSONObject(builder.toString()); 
196      }
