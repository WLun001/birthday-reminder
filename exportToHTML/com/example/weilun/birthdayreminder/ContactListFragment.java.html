130      @Override 
131      public void onActivityResult(int requestCode, int resultCode, Intent data) { 
132          super.onActivityResult(requestCode, resultCode, data); 
133          //result code == -1 when positive button, else is 0 
134          if (resultCode == -1) { 
135              PersonDBQueries dbQueries = new PersonDBQueries(new PersonDBHelper(getActivity())); 
136              dbQueries.deleteAll(); 
137              getLoaderManager().restartLoader(SEARCH_LOADER_ID, null, this); 
138              refreshable.onRefresh(); 
139              Toast.makeText(getActivity(), R.string.delete_success, Toast.LENGTH_SHORT).show(); 
140          }

157      public boolean onQueryTextChange(String query) { 
158          if (!TextUtils.isEmpty(query)) { 
159              searchKeyword = query;

176      @Override 
177      public void onLoadFinished(Loader<Cursor> loader, Cursor data) { 
178          loadingBar.setVisibility(View.GONE); 
179          tv.setText(getString(R.string.no_birthday_data)); 
180          Log.v("onLoadFinish", Integer.toString(data.getColumnCount()));

183          Log.v("onLoadFinish", "Swapping adapter");

200      /** 
201       * A class to perform search in background 
202       */ 
203      public static final class SearchLoader extends AsyncTaskLoader<Cursor> { 
204          private String keyword; 
205          private Context context; 
206   
207          public SearchLoader(Context context, String keyword) { 
208              super(context); 
209              Log.v("SearchLoader", "SearchLoader instantiated"); 
210              this.context = context; 
211              this.keyword = keyword; 
212              Log.v("SearchLoader", "Keyword value: " + keyword); 
213          } 
214   
215          @Override 
216          protected void onStartLoading() { 
217              super.onStartLoading(); 
218              forceLoad(); 
219          } 
220   
221          @Override 
222          public Cursor loadInBackground() { 
223              PersonDBQueries dbQuery = new PersonDBQueries(new PersonDBHelper(context)); 
224              Cursor cursor; 
225              String[] columns = PersonContract.columns; 
226              if (keyword != null) { 
227                  Log.v("loadInBackground", "LIKE QUERY"); 
228                  String[] selectionArgs = {"%" + keyword + "%"}; 
229                  cursor = dbQuery.query(columns, PersonContract.PersonEntry.COLUMN_NAME_NAME + " LIKE ?" 
230                          , selectionArgs, null, null 
231                          , null); 
232              } else { 
233                  Log.v("loadInBackgrond", "query in background"); 
234                  cursor = dbQuery.query(columns, null, null, null, null 
235                          , PersonContract.PersonEntry.COLUMN_NAME_NAME + " ASC"); 
236              } 
237              return cursor; 
238          } 
239      }
